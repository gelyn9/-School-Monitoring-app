<script>
const students = [
 {id:1,name:"Emma Johnson",grade:"12th",status:"present",avg:92},
 {id:2,name:"Liam Smith",grade:"11th",status:"present",avg:88},
 {id:3,name:"Olivia Brown",grade:"10th",status:"late",avg:85},
 {id:4,name:"Noah Davis",grade:"12th",status:"present",avg:90},
 {id:5,name:"Ava Wilson",grade:"11th",status:"absent",avg:78}
];

const classes = [
 "Mathematics - Room 101",
 "Physics - Lab 205",
 "English - Room 103",
 "Chemistry - Lab 206"
];

const teachers = [
 "Dr. Sarah Johnson",
 "Prof. Michael Chen",
 "Ms. Emily Davis",
 "Dr. Robert Lee"
];

function showTab(id){
 document.querySelectorAll('.tab').forEach(t=>t.classList.remove('active'));
 document.getElementById(id).classList.add('active');
}

function updateDashboard(){
 totalStudents.textContent = students.length;
 presentToday.textContent = students.filter(s=>s.status==="present").length;
 averageGrade.textContent = Math.round(students.reduce((a,s)=>a+s.avg,0)/students.length);
 totalClasses.textContent = classes.length;
}

function renderStudents(){
 studentTable.innerHTML="";
 students.forEach(s=>{
  studentTable.innerHTML+=`<tr><td>${s.name}</td><td>${s.grade}</td><td>${s.status}</td><td>${s.avg}</td></tr>`;
 });
}

function renderAttendance(){
 attendanceList.innerHTML="";
 students.forEach(s=>{
  attendanceList.innerHTML+=`
   <li>${s.name}
    <select onchange="s.status=this.value;updateDashboard()">
     <option ${s.status==="present"?"selected":""}>present</option>
     <option ${s.status==="absent"?"selected":""}>absent</option>
     <option ${s.status==="late"?"selected":""}>late</option>
    </select>
   </li>`;
 });
}

classes.forEach(c=>classList.innerHTML+=`<li>${c}</li>`);
teachers.forEach(t=>teacherList.innerHTML+=`<li>${t}</li>`);

updateDashboard();
renderStudents();
renderAttendance();

new Chart(attendanceChart,{
 type:"bar",
 data:{
  labels:["Mon","Tue","Wed","Thu","Fri"],
  datasets:[
   {label:"Present",data:[145,142,148,140,147],backgroundColor:"#1976d2"},
   {label:"Absent",data:[5,8,2,10,3],backgroundColor:"#e53935"}
  ]
 }
});

new Chart(performanceChart,{
 type:"line",
 data:{
  labels:["Math","Science","English","History","Arts"],
  datasets:[{label:"Average Score",data:[85,88,82,86,90],borderColor:"#1976d2"}]
 }
});
</script>
